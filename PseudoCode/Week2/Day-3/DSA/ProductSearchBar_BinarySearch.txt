
// Pseudocode: Product Search Bar using Binary Search

function binarySearch(array, target):
    left = 0
    right = array.length - 1

    while left <= right:
        mid = (left + right) / 2
        if array[mid] == target:
            return mid
        else if array[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return -1

main():
    catalog = ["AirPods", "Backpack", "Camera", "Laptop", "Shoes", "Smartphone", "Watch"]
    search1 = binarySearch(catalog, "Laptop") // found
    search2 = binarySearch(catalog, "Headphones") // not found
    search3 = binarySearch(catalog, "Shoes") // found
